\textbf{Global setup.} The paper studies a piecewise low-rank bandit:
$\theta_t = B_k^\star w_t$ for $t \in \cI_k$, where $B_k^\star \in \R^{d\times r}$
has orthonormal columns (piecewise constant across $K$ segments),
$w_t$ follows a stable LDS with $\rho(A_k) \le 1-\alpha_0$,
$\|w_t\| \le S_w$. The learner plays $x_t \in \cA_t$ with $\|x\|\le R_\cA$
and observes $y_t = x_t^\top \theta_t + \varepsilon_t$ with sub-Gaussian
noise, variance $\sigma_\varepsilon^2$ (estimated: $\hat\sigma^2$).
Probes are isotropic Gaussian $u_t \sim \mathcal N(0, I_d)$, truncated
at $\|u\|\le L$. Costed dynamic regret:
$\DynReg_T^{(c)} := \sum_t(\max_{x\in\cA_t}x^\top\theta_t - x_t^\top\theta_t) + c|\cTprobe|$.

Key derived quantities used across theorems:
\begin{itemize}[leftmargin=*,itemsep=1pt,topsep=1pt]
  \item $\varepsilon_k := \|\widehat P_k - P_k^\star\|_{\op}$: subspace error.
  \item $\bar M_k^{\mathrm{probe}} := m_k^{-1}\sum_{t\in\cT_k}\widetilde M_t$
        where $\widetilde M_t := \E[\theta_t\theta_t^\top|\cH_{t-1}]$.
  \item $V := \sum_k\sum_{s\in E_k, s+1\in E_k}\|\theta_{s+1}-\theta_s\|_2$:
        path variation.
  \item $L_W := \log(1+WR_\cA^2/(\lambda r))$.
  \item $R_X := L^2 R_s + S_w^2$, $R_s := L_y^2 + \hat\sigma^2$,
        $L_y := LS_w + L_\varepsilon$, $L_\varepsilon := \sigma_\varepsilon\sqrt{2\log(4T/\delta)}$.
  \item $\Delta_\sigma := 4(|\hat\sigma^2-\sigma_\varepsilon^2|L^2 + 2L^3\epsilon_\times)/\lambda_{\min}$:
        bias floor (conservative, any probe distribution).
  \item $\Delta_\sigma^{\mathrm{sharp}} := 8L^3\epsilon_\times/\lambda_{\min}$:
        bias floor (sharp, Gaussian probes only; scaled-identity
        $\delta_\sigma$-bias does not rotate eigenvectors).
  \item $C_{\mathrm{sub}} := 8R_X/\lambda_{\min}$: Davis--Kahan constant.
  \item $\Pi_{\mathrm{HP}}$: class of policies with $x_t = \widehat P_t x_t$,
        $\widehat P_t$ rank-$r$ data-measurable projector.
\end{itemize}

\section{Concentration chain (from chunk A2)}

\begin{lemma}[Observable a.s.\ bound on $G_t$]
\label{lem:G_bound}
Define $\cA_t := \{\|u_t\|_2\le L\}\cap\{|y_t|\le L_y\}$. Both indicators
are $\sigma(u_t,y_t)$-measurable. On $\cA_t$, $|s_t|\le R_s$ and
$\|G_t\|_\op\le R_X$. With $L^2=2d\log(4T/\delta)$ and
$L_\varepsilon=\sigma_\varepsilon\sqrt{2\log(4T/\delta)}$,
$\Pr(\cA_t^c\mid\cH_{t-1})\le\delta/T$, hence
$\Pr(\bigcap_{t\in\cTprobe}\cA_t)\ge 1-\delta$.
\end{lemma}

\begin{lemma}[Lifted probe sample, scaled-identity bias]
\label{lem:G_unbiased}
For Gaussian probes, $\E[G_t\mid\cH_{t-1}] = \widetilde M_t + \widetilde B_t$
with $\widetilde B_t = -\delta_\sigma/(d+2)\,I_d$ (scaled identity,
exact; Gaussian odd-moment vanishing kills the $\epsilon_\times$ piece).
Hence $\widetilde B_t$ does not rotate eigenvectors.
\end{lemma}

\begin{theorem}[Matrix Freedman for $\widehat M_k$ via predictable truncation]
\label{thm:matrix_bernstein}
Define $\tilde G_t := G_t\cdot\mathbf 1\{\cA_t\}$ and
$\tilde X_t := \tilde G_t - \E[\tilde G_t\mid\cH_{t-1}]$.
Then $\{\tilde X_t\}$ is an MDS with $\|\tilde X_t\|_\op\le 2R_X$ a.s.
and $\|\E[\tilde X_t^2\mid\cH_{t-1}]\|_\op\le R_X^2$. With probability
at least $1-2\delta$,
\[
\|\widehat M_k - \bar M_k^{\mathrm{probe}} - \widetilde B\|_\op
\le 2R_X\sqrt{\log(2d/\delta)/m_k} + \frac{2R_X\log(2d/\delta)}{3m_k}
+ \|\Theta_k\|_\op,
\]
where $\Theta_k$ is the truncation-bias offset
(bounded in Prop.\ \ref{prop:theta_bound}).
\end{theorem}

\begin{corollary}[Projector concentration]
\label{cor:projector}
Assume $m_k$ is large enough that both (i)
$\|\widehat M_k - \bar M_k^{\mathrm{probe}}\|_\op\le\lambda_{\min}/2$
and (ii) $m_k\ge\log(2d/\delta)$. Then with probability $\ge 1-2\delta$,
$\varepsilon_k \le C_{\mathrm{sub}}\sqrt{\log(2d/\delta)/m_k} + \Delta_\sigma$.
For Gaussian probes, $\Delta_\sigma$ specializes to the sharper
$\Delta_\sigma^{\mathrm{sharp}} = 4|\delta_\sigma|/((d+2)\lambda_{\min})$.
\end{corollary}

\section{Theorem 1 --- Main upper bound (from chunk A4)}

\begin{theorem}[Costed dynamic regret of SPSC]
\label{thm:spsc_regret}
Run SPSC with probe allocation $m_k = \lceil (B\ell_k/(2A))^{2/3}\rceil$,
where $A := 2R_\cA S_w + c$ and
$B := 2 C_{\mathrm{sub}} S_w R_\cA \sqrt{\log(d/\delta)}$.
With probability at least $1-\delta$,
\[
\DynReg_T^{(c)}
= \tilO\!\bigl(r\sqrt{K T L_W}\bigr)
+ \tilO\!\bigl(K^{1/3}T^{2/3}\bigr)
+ O(W\,V)
+ O(T\,\Delta_\sigma).
\]
The bound is informative precisely when
\textbf{(a)} $K=o(\min\{T^{1/2}/r^{3/2},d^2/r^2\})$,
\textbf{(b)} $V=o(r\sqrt{KT}/W)$, and
\textbf{(c)} $\Delta_\sigma = o(1)$.
(b) holds in the piecewise-constant regime ($V\le 2KS_w$) but fails
for stationary stable LDS ($V=\Theta(T)$). (c) holds under exact
probes $\delta_\sigma=\epsilon_\times=0$.
\end{theorem}

\begin{lemma}[Windowed ridge prediction bound]
\label{lem:drift}
For the projected ridge with residual $\xi_s$ bounded by $\bar\xi_k$:
\[
|(\widehat a_t - a_t^\star)^\top z|
\le \beta_t^{(r,W)}\|z\|_{\widetilde V_t^{-1}}
+ R_\cA V_{k,t}(W)
+ \bar\xi_k\sqrt{2W r L_W}\,\|z\|_{\widetilde V_t^{-1}}.
\]
For SPSC, $\bar\xi_k=R_\cA S_w\varepsilon_k$ via the equal-rank
projector identity $\|(I-\widehat P)B^\star\|_\op\le\|P^\star-\widehat P\|_\op$.
\end{lemma}

\section{Theorem 2 --- Minimax subspace-recovery LB (from chunk A3)}

\begin{theorem}[Minimax subspace-recovery lower bound]
\label{thm:minimax}
Let $d\ge 2r$. There exists a constant $c_1=\tfrac{1}{32}\sqrt{c_0(d-r)/C_1}$
($C_1 := 8\lambda_{\min}^2/\sigma_\varepsilon^4$) such that for every
$m\ge c_0(d-r)/(16 C_1)$ and every $\cH_m$-measurable projector
estimator $\widehat P$,
\[
\inf_{\widehat P}\sup_{\nu\in\mathfrak M}\E_\nu[\|\widehat P - P^\star\|_\op]
\ge c_1/\sqrt m.
\]
Matched by Cor.\ projector\_conf up to $\sqrt{\log d}$.
\end{theorem}

\begin{corollary}[Identifiability sample threshold]
\label{cor:id_threshold}
Any estimator with op-norm subspace error below $1/32$ requires
$m \ge c_{\mathrm{id}}(d-r)$ probe rounds.
\end{corollary}

\section{Theorem 3 --- Hard-projection regret LB (from chunk A7)}

\begin{theorem}[Lower bound within $\Pi_{\mathrm{HP}}$]
\label{thm:regret_lb}
Assume $d \ge d_0\cdot r\,T^{1/3}$ for a constant $d_0$ (large-$d$ regime).
For all sufficiently large $T$, there exists $c_0 > 0$ (depending on
$R_\cA, \sigma_\varepsilon, d_0$) such that
\[
\inf_{\pi\in\Pi_{\mathrm{HP}}} \sup_\nu\ \E^\pi_\nu[\DynReg_T^{(c)}]
\ge c_0\,c^{1/3}(R_\cA\sigma_\varepsilon)^{2/3} T^{2/3}.
\]
In the opposite regime $d=O(r)$, the Pinsker bound degrades to $\Omega(\sqrt T)$.
\end{theorem}

\section{Corollary --- Rank-adaptive (from chunk A5)}

\begin{corollary}[Rank-adaptive guarantee]
\label{cor:rank_adaptive}
Assume exact probes or Gaussian probes. If $r$ is unknown, thresholding
$\widehat M_k$ at $\tau_k^\star := 2R_X\sqrt{\log(2d/\delta)/m_k}+|\widetilde B|$
(where $|\widetilde B|:=|\delta_\sigma|/(d+2)$ is known to the algorithm
from plugin $\hat\sigma^2$) recovers the true rank w.p.\ $\ge 1-\delta$
whenever $\lambda_{\min}\ge 4\tau_k^\star$, and Thm.\ \ref{thm:spsc_regret}
applies unchanged (with extra $\delta$ in the union bound).
\end{corollary}

\section{Theorem --- SPSC-Adaptive (from chunk A6)}

\begin{theorem}[Adaptive SPSC]
\label{thm:spsc_adaptive}
Suppose every true subspace change satisfies $\Delta_k := \|P_{k+1}^\star - P_k^\star\|_\op \ge 2b$.
Run SPSC-Adaptive with Bernoulli probe rate $\mu = c_\mu\,(K/T)^{1/3}$
(tuning constant $c_\mu > 0$). Then with probability $\ge 1-\delta-\delta_{\mathrm{FA}}$,
\[
\DynReg_T^{(c)} = \tilO(r\sqrt{KT}) + \tilO(K^{1/3}T^{2/3}) + O(WV) + O(T\Delta_\sigma) + O(KW_{\mathrm{det}}/\mu).
\]
The scaling $\mu\asymp(K/T)^{1/3}$ is necessary, not cosmetic: a
universal constant $\mu\in(0,1)$ produces probe cost $\Theta(T)$,
blowing up the rate by $T^{1/3}$.
\end{theorem}

\section{Corollary --- Approximate low rank (from chunk cor\_misspec)}

\begin{corollary}[Robustness to approximate low rank]
\label{cor:misspec}
Suppose $\theta_t = B_k^\star w_t + \theta_t^{\perp}$ with
$\sup_{t\in\cI_k}\|\theta_t^\perp\|_2\le\epsilon_k^\perp \le S_w$.
Assume the strengthened eigengap condition
$\lambda_{\min}\ge 4 C_{\mathrm{sub}}\sqrt{\log(2d/\delta)/m_k} + 12\,S_w\,\epsilon_k^\perp$.
Then Thm.\ \ref{thm:spsc_regret} holds with the additive term
\[
O\!\Bigl(R_\cA\sum_k \epsilon_k^\perp\,\bigl(1 + S_w^2/\lambda_{\min}\bigr)\, n_k\Bigr).
\]
\end{corollary}

\begin{remark}[Variance misspecification: two bounds]
\label{rem:variance_misspec}
Write $\delta_\sigma:=\hat\sigma^2-\sigma_\varepsilon^2$. Thm.\ \ref{thm:spsc_regret}
uses the CONSERVATIVE $\Delta_\sigma$ (includes $|\delta_\sigma|L^2/\lambda_{\min}$),
valid for any probe distribution. For ISOTROPIC Gaussian probes,
the $\delta_\sigma$-component of the bias is a SCALED IDENTITY
$-\delta_\sigma/(d+2)\cdot I_d$, which does not rotate eigenvectors;
hence a SHARPER bound
$\Delta_\sigma^{\mathrm{sharp}}:=8L^3\epsilon_\times/\lambda_{\min}$
(dropping the $|\delta_\sigma|L^2$ part) applies, via a tightened
eigengap condition. We always have $\Delta_\sigma^{\mathrm{sharp}} \le \Delta_\sigma$.
\end{remark}

\section{Appendix C --- Truncation (from chunk truncation)}

\begin{proposition}[Truncation-bias bound]
\label{prop:theta_bound}
The offset $\Theta_k := m_k^{-1}\sum_{t\in\cT_k}\E[G_t\mathbf 1\{\cA_t^c\}\mid\cH_{t-1}]$
satisfies $\|\Theta_k\|_\op \le O(d\sqrt{\delta/T})$, polynomially small
in $T$ for $\delta$ polynomial in $1/T$, and dominated by the Freedman
bound for $m_k = O(T^{2/3})$. Proof uses explicit Gaussian moments
$\E[(u^\top\theta)^8]=105\|\theta\|^8$ and binomial
$(a+b+c)^4\le 27(a^4+b^4+c^4)$ to control $\E[\|G_t\|_\op^2\mid\cH_{t-1}]=O(d^2)$.
\end{proposition}
