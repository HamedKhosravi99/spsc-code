\textbf{Setup context.}
\begin{itemize}[leftmargin=*,itemsep=1pt,topsep=1pt]
\item Piecewise low-rank: $\theta_t=B_k^\star w_t$ on $\cI_k$,
$B_k^\star\in\R^{d\times r}$ orthonormal, $\|w_t\|\le S_w$; actions
$\|x\|\le R_\cA$; sub-Gaussian noise variance $\sigma_\varepsilon^2$
(plugin $\hat\sigma^2$, $\delta_\sigma:=\hat\sigma^2-\sigma_\varepsilon^2$).

$\varepsilon_k\le C_{\mathrm{sub}}\sqrt{\log(2d/\delta)/m_k}+\Delta_\sigma$
with $\Delta_\sigma := 4(|\delta_\sigma|L^2+2L^3\epsilon_\times)/\lambda_{\min}$
and $C_{\mathrm{sub}}:=8R_X/\lambda_{\min}$, $R_X:=L^2 R_s+S_w^2$.
\item Windowed ridge on projected covariates $z_t=\widehat U_{t-1}^\top x_t\in\R^r$
with $L_W=\log(1+WR_\cA^2/(\lambda r))$; F2 / F3 standard.
\end{itemize}

\section*{Statement of Theorem 1}

\begin{theorem}[Costed dynamic regret of SPSC]
\label{thm:spsc_regret}
Run SPSC with probe allocation $m_k = \lceil (B\ell_k/(2A))^{2/3}\rceil$,
where $A := 2R_\cA S_w + c$ and
$B := 4 C_{\mathrm{sub}} S_w R_\cA \sqrt{\log(2d/\delta)}$.
With probability $\ge 1-\delta$,
\begin{equation}
\DynReg_T^{(c)}
= \tilO\!\bigl(r\sqrt{K T L_W}\bigr)
+ \tilO\!\bigl(K^{1/3}T^{2/3}\bigr)
+ O(W\,V)
+ O(T\,\Delta_\sigma).
\label{eq:main_bound}
\end{equation}
The bound is informative precisely when
\textbf{(a)} $K=o(\min\{T^{1/2}/r^{3/2},d^2/r^2\})$,
\textbf{(b)} $V=o(r\sqrt{KT}/W)$ (which holds in the piecewise-constant
regime $V\le 2KS_w$ but fails for stationary stable LDS where $V=\Theta(T)$),
and \textbf{(c)} $\Delta_\sigma = o(1)$ (holds under exact probes
$\delta_\sigma=\epsilon_\times=0$).
\end{theorem}

\section*{Proof}

Fix segment $\cI_k$ with probe budget $m_k$ and let
$\Delta_t:=\max_{x\in\cA_t}x^\top\theta_t - x_t^\top\theta_t$ denote
instantaneous exploit regret. Decompose costed regret into three pieces.

\subsection*{(i) Projected exploitation regret}

The learner plays
$x_t=\argmax_{x\in\cA_t}\mathrm{UCB}(x)$ where
$\mathrm{UCB}(x) := (\widehat U_{t-1}^\top x)^\top\widehat a_t
+ \widetilde\beta_t^{(r,W)}\|\widehat U_{t-1}^\top x\|_{\widetilde V_t^{-1}}
+ \gamma_t\|x\|_2$
(using the \emph{inflated} radius $\widetilde\beta_t^{(r,W)}$ so that
optimism covers the full Lem.~\ref{lem:drift_conf} bound).
Write $a_t^\star:=\widehat U_{t-1}^\top\theta_t$. For
$x_t^\star\in\argmax_x x^\top\theta_t$, when the UCB event holds,
optimism gives
$x_t^{\star\top}\theta_t\le\mathrm{UCB}(x_t^\star)\le\mathrm{UCB}(x_t)$
(the second by $x_t=\argmax\mathrm{UCB}$), hence
\begin{align*}
\Delta_t
&\le \mathrm{UCB}(x_t) - x_t^\top\theta_t \\
&= x_t^\top(\widehat U_{t-1}\widehat a_t - \theta_t)
 + \widetilde\beta_t^{(r,W)}\|\widehat U_{t-1}^\top x_t\|_{\widetilde V_t^{-1}}
 + \gamma_t\|x_t\|_2 \\
&\le \bigl|x_t^\top\widehat U_{t-1}(\widehat a_t - a_t^\star)\bigr|
 + \bigl|x_t^\top(I - \widehat P_{t-1})\theta_t\bigr|
 + \widetilde\beta_t^{(r,W)}\|\widehat U_{t-1}^\top x_t\|_{\widetilde V_t^{-1}}
 + \gamma_t\|x_t\|_2.
\end{align*}
Line~2 uses $\widehat U_{t-1}\widehat a_t = \widehat P_{t-1}\widehat U_{t-1}\widehat a_t$
and $\widehat P_{t-1}\theta_t = \widehat U_{t-1}a_t^\star$, so
$\widehat U_{t-1}\widehat a_t-\theta_t = \widehat U_{t-1}(\widehat a_t-a_t^\star)-(I-\widehat P_{t-1})\theta_t$;
line~3 applies the triangle inequality, splitting into two non-negative
pieces bounded above. We choose $\gamma_t := S_w\varepsilon_{k,t}$,
which produces the \emph{pointwise} domination
$|x^\top(I-\widehat P_{t-1})\theta|\le\|x\|_2\|(I-\widehat P_{t-1})\theta\|_2\le\|x\|_2 S_w\varepsilon_k=\gamma_t\|x\|_2$
(Cauchy--Schwarz + equal-rank projector identity), so the $\gamma_t\|x\|_2$
slack absorbs the subspace-mismatch bias at every $x$, ensuring optimism
holds without requiring $R_\cA\ge 1$.

\emph{(i.a) Subspace-mismatch term.}
$|x_t^\top(I-\widehat P_{t-1})\theta_t|\le R_\cA\|(I-\widehat P_{t-1})B_k^\star\|_\op\|w_t\|
\le R_\cA S_w\varepsilon_k$ by the equal-rank projector identity
$\|(I-\widehat P)B^\star\|_\op\le\|P^\star-\widehat P\|_\op$.
Summed over $E_k$: $R_\cA S_w\varepsilon_k n_k$.

\emph{(i.b) In-subspace estimation.} The projected ridge is
\emph{misspecified}: residual $y_s-z_s^\top a_s^\star = \xi_s+\varepsilon_s$
with $|\xi_s|\le\bar\xi_k:=R_\cA S_w\varepsilon_k$. We apply
Lem.~\ref{lem:drift_conf} (below): for $z=z_t(x_t)$ with $\|z\|\le R_\cA$,
\[
|(\widehat a_t-a_t^\star)^\top z|
\le\bigl(\beta_t^{(r,W)}+R_\cA V_{k,t}(W)\sqrt{W}+R_\cA S_w\varepsilon_k\sqrt{2WrL_W}\bigr)\|z\|_{\widetilde V_t^{-1}}.
\]
Summing over $E_k$: noise and mismatch-propagation handled by F3
($\sum_{t\in E_k}\|z_t\|_{\widetilde V_t^{-1}}\le\sqrt{2rn_kL_W}$); drift
handled by Cauchy--Schwarz in $t$ together with F3 and
$\sum_t V_{k,t}(W)^2\le(\max_t V_{k,t}(W))\sum_t V_{k,t}(W)\le V_k\cdot WV_k=WV_k^2$
(each drift edge in $\le W$ windows):
\begin{equation}
\sum_{t\in E_k}|(\widehat a_t-a_t^\star)^\top z_t|
\le\beta_k^{(r,W)}\sqrt{2rn_kL_W}+R_\cA W V_k\sqrt{2rL_W}
+2rL_W R_\cA S_w\varepsilon_k\sqrt{Wn_k}.
\label{eq:in_subspace_sum}
\end{equation}
The third term (mismatch propagation through ridge) is lower-order
than $R_\cA S_w\varepsilon_k n_k$ when $n_k\ge W(2rL_W)^2$, absorbed
into $\tilO(\cdot)$.

\emph{(i.c) Total exploit regret.}
Distinguish the \emph{base statistical radius} $\beta_t^{(r,W)}$ (from F2)
and the \emph{algorithmic inflated radius}
$\widetilde\beta_t^{(r,W)} := \beta_t^{(r,W)}+R_\cA V_{k,t}(W)\sqrt{W}+R_\cA S_w\varepsilon_k\sqrt{2WrL_W}$
used in the UCB; the latter dominates the full
Lem.~\ref{lem:drift_conf} bound (standard Russac sliding-window
convention). The $\gamma_t\|x\|$ slack with
$\gamma_t := S_w\varepsilon_{k,t}$ produces the pointwise domination
$|x^\top(I-\widehat P)\theta|\le\|x\|\|(I-\widehat P)\theta\|\le\|x\|S_w\varepsilon_k=\gamma_t\|x\|$
(Cauchy--Schwarz + equal-rank projector identity), which covers the
subspace-mismatch bias at every $x$ with no $R_\cA\ge 1$ assumption. Adding (i.a), (i.b), and the $\gamma_t\|x_t\|$
accounting (contributes $\sum_t\gamma_t\|x_t\|\le R_\cA S_w\varepsilon_k n_k$):
\begin{equation}
\sum_{t\in E_k}\Delta_t
\le 2\beta_k^{(r,W)}\sqrt{2rn_kL_W}+2R_\cA S_w\varepsilon_k n_k
+2R_\cA WV_k\sqrt{2rL_W}+(\text{l.o.}).
\label{eq:exploit_sum}
\end{equation}
Summing over $K$ segments via $\sum_k\sqrt{n_k}\le\sqrt{KT}$
(Cauchy--Schwarz) and $\sum_k V_k=V$:
$\tilO(r\sqrt{KT})+\sum_k 2R_\cA S_w\varepsilon_k n_k+O(R_\cA W V)$. The
factor-2 is absorbed into $B:=4 C_{\mathrm{sub}} S_w R_\cA\sqrt{\log(2d/\delta)}$
in step (ii) below (not $2C_{\mathrm{sub}}\cdots$, the R3 coefficient).

\subsection*{(ii) Probe--subspace tradeoff}

Cor.~projector\_conf at confidence $\delta/(2K)$ gives
$\varepsilon_k\le C_{\mathrm{sub}}\sqrt{\log(4Kd/\delta)/m_k}+\Delta_\sigma$.
Substituting into the (factor-2) step-(i.c) term $2R_\cA S_w\varepsilon_k n_k$:
\[
2R_\cA S_w\varepsilon_k n_k\le B n_k/\sqrt{m_k}+2R_\cA S_w\Delta_\sigma n_k,
\qquad B:=4C_{\mathrm{sub}} S_w R_\cA\sqrt{\log(2d/\delta)}.
\]
Probe cost $Am_k$, $A:=2R_\cA S_w+c$. Minimising
$Am_k+B\ell_k/\sqrt{m_k}$ in $m_k\in[1,\ell_k]$:
$m_k^\star=\lceil(B\ell_k/(2A))^{2/3}\rceil$ and
\begin{equation}
A m_k^\star+\frac{B\ell_k}{\sqrt{m_k^\star}}
=\tfrac{3}{2^{2/3}}A^{1/3}B^{2/3}\ell_k^{2/3}(1+o(1)).
\label{eq:opt_mk}
\end{equation}
Summing, Jensen $\sum_k\ell_k^{2/3}\le K^{1/3}T^{2/3}$ gives
$\tilO(K^{1/3}T^{2/3})$. The bias-floor piece sums to
$O(T\Delta_\sigma)$; under exact probes it vanishes.

\subsection*{(iii) Within-segment drift}

\begin{lemma}[Windowed ridge prediction bound]
\label{lem:drift_conf}
Assume the projected-regression observation model on $s\in\cW_t$ is
\begin{equation}
y_s = z_s^\top a_s^\star + \xi_s + \varepsilon_s,\qquad
|\xi_s|\le\bar\xi_k,\qquad \varepsilon_s\text{ F2-sub-Gaussian},
\label{eq:ridge_residual_decomp}
\end{equation}
where the projected covariate $z_s:=\widehat U_{t-1}^\top x_s$ and the
target $a_s^\star:=\widehat U_{t-1}^\top\theta_s$ both use the
\emph{single reference projector} $\widehat U_{t-1}$ across the window
$\cW_t$ (standard Russac convention: ridge inside a window is solved
with respect to a fixed basis, the one available at the end of the window).
$\xi_s$ is a \emph{deterministic residual} (subspace-mismatch bias), and
$\varepsilon_s$ is sub-Gaussian. On the F2 event, the windowed ridge
$\widehat a_t=\widetilde V_t^{-1}\sum_{s\in\cW_t}z_sy_s$ satisfies
for every $\|z\|\le R_\cA$,
\begin{equation}
|(\widehat a_t-a_t^\star)^\top z|
\le\bigl(\beta_t^{(r,W)}+R_\cA V_{k,t}(W)\sqrt{W}+\bar\xi_k\sqrt{2WrL_W}\bigr)\|z\|_{\widetilde V_t^{-1}},
\label{eq:drift_lemma_bound}
\end{equation}
where $V_{k,t}(W):=\sum_{s\in\cW_t,s+1\in\cW_t}\|a_{s+1}^\star-a_s^\star\|_2$.
\end{lemma}
\begin{proof}
Write $\eta_s:=a_s^\star-a_t^\star$. The ridge normal equations give
\[
\widetilde V_t(\widehat a_t-a_t^\star)
=\sum_{s\in\cW_t}z_s(z_s^\top\eta_s+\xi_s+\varepsilon_s)-\lambda a_t^\star
=\underbrace{\sum z_sz_s^\top\eta_s}_{B_t^{\mathrm{drift}}}
+\underbrace{\sum z_s\xi_s}_{B_t^{\mathrm{sub}}}
+\underbrace{\sum z_s\varepsilon_s-\lambda a_t^\star}_{N_t}.
\]
\emph{Noise.} Cauchy--Schwarz in $\widetilde V_t^{-1}$-inner product +
F2 give $|z^\top\widetilde V_t^{-1}N_t|\le\|z\|_{\widetilde V_t^{-1}}\beta_t^{(r,W)}$.

\emph{Drift.} By Cauchy--Schwarz in the $\widetilde V_t^{-1}$-inner
product, $|z^\top\widetilde V_t^{-1}B_t^{\mathrm{drift}}|
\le\|z\|_{\widetilde V_t^{-1}}\|B_t^{\mathrm{drift}}\|_{\widetilde V_t^{-1}}$.
Set $c_s := z_s^\top\eta_s$; by telescoping $\|\eta_s\|\le V_{k,t}(W)$
uniformly, so $|c_s|\le R_\cA V_{k,t}(W)$. Writing
$B_t^{\mathrm{drift}}=\sum_s c_s z_s$ and expanding the matrix inner product,
\[
\|B_t^{\mathrm{drift}}\|^2_{\widetilde V_t^{-1}}
= \sum_{s,s'\in\cW_t} c_s c_{s'} z_s^\top\widetilde V_t^{-1}z_{s'}
= c^\top M c,
\]
where $M_{ss'}:=z_s^\top\widetilde V_t^{-1}z_{s'}$ is the Gram matrix of
$\{\widetilde V_t^{-1/2}z_s\}_{s\in\cW_t}$. Since
$\sum_s z_sz_s^\top\preceq\widetilde V_t$, we have
$M = Z\widetilde V_t^{-1}Z^\top \preceq I_{|\cW_t|}$ ($Z$ stacking the
$z_s$ as rows), so $c^\top M c\le\|c\|_2^2\le W R_\cA^2 V_{k,t}(W)^2$.
Hence $\|B_t^{\mathrm{drift}}\|_{\widetilde V_t^{-1}}\le R_\cA V_{k,t}(W)\sqrt W$
and $|z^\top\widetilde V_t^{-1}B_t^{\mathrm{drift}}|\le\|z\|_{\widetilde V_t^{-1}}\,R_\cA V_{k,t}(W)\sqrt W$.
(The R2 argument via $\|H_u\|_\op\le 1$ for $H_u=\widetilde V_t^{-1}\sum_{s<u}z_sz_s^\top$
was invalid: similarity preserves spectrum, not operator norm.)

\emph{Mismatch propagation.} Cauchy--Schwarz:
$\|B_t^{\mathrm{sub}}\|_{\widetilde V_t^{-1}}\le\bar\xi_k\sum_s\|z_s\|_{\widetilde V_t^{-1}}
\le\bar\xi_k\sqrt{W\cdot 2rL_W}$ by F3 and $\sum_s\|z_s\|_{\widetilde V_t^{-1}}\le\sqrt W\sqrt{\sum_s\|z_s\|_{\widetilde V_t^{-1}}^2}$.
Hence $|z^\top\widetilde V_t^{-1}B_t^{\mathrm{sub}}|\le\|z\|_{\widetilde V_t^{-1}}\cdot\bar\xi_k\sqrt{2WrL_W}$.
Summing the three terms yields~\eqref{eq:drift_lemma_bound}.
\end{proof}

\begin{remark}[Specialization to SPSC]
\label{rem:drift_lemma_spec}
For SPSC's projected ridge, the residual is
$\xi_s:=x_s^\top(I-\widehat P_{t-1})\theta_s$, using the \emph{same}
reference projector $\widehat P_{t-1}$ that Lem.~\ref{lem:drift_conf}
uses for $z_s$. (Inside a segment's exploitation window,
$\widehat P_{t-1}=\widehat P_{s-1}=\widehat P_k$ is constant, so the
two conventions coincide.)
$|\xi_s|\le R_\cA\|(I-\widehat P_{t-1})B_k^\star\|_\op S_w\le R_\cA S_w\varepsilon_k$
(equal-rank identity). Hence $\bar\xi_k=R_\cA S_w\varepsilon_k$ in
\eqref{eq:drift_lemma_bound}.
\end{remark}

\emph{Mismatch-propagation magnitude.} Plugging
$m_k^\star\asymp T^{2/3}K^{-2/3}$, $\varepsilon_k\asymp K^{1/3}T^{-1/3}$,
$n_k\asymp T/K$ into the propagation term of \eqref{eq:in_subspace_sum}:
$2rL_W R_\cA S_w\varepsilon_k\sqrt{Wn_k}\asymp\tilO(W^{1/2}K^{-1/6}T^{1/6})$
per segment, summing to $\tilO(K^{5/6}W^{1/2}T^{1/6})$, which is
$o(K^{1/3}T^{2/3})$ whenever $WK=o(T)$.

\emph{Drift summation.}
$\|a_{s+1}^\star-a_s^\star\|\le\|\theta_{s+1}-\theta_s\|$ appears in
$\le W$ windows, so $\sum_{k,t}V_{k,t}(W)\le WV$ and total drift is
$O(R_\cA W V)$.

\emph{Regime of interest.} Informative iff $V=o(r\sqrt{KT}/W)$. For
piecewise constant: $V\le 2KS_w$ deterministically; for $W\asymp\sqrt{T/K}$
drift is $O(\sqrt{KT}R_\cA S_w)$, lower order. For stationary stable
LDS, $V=\Theta(T)$, so $O(WV)$ dominates.

Union bound over $K$ segment events ($\delta/(2K)$) + windowed
self-normalised event ($\delta/2$) gives, with probability $\ge 1-\delta$,
\[
\DynReg_T^{(c)} = \tilO(r\sqrt{KT})+\tilO(K^{1/3}T^{2/3})+O(WV)+O(T\Delta_\sigma).
\qedhere
\]

\begin{remark}[Robustness to segment-length misestimation]
\label{rem:lk_misspec}
If $\widehat\ell_k\in[\ell_k/c_\ell,c_\ell\ell_k]$, per-segment
probe--subspace cost degrades by at most
$\max(c_\ell^{2/3},c_\ell^{-1/3})$; the leading $K^{1/3}T^{2/3}$
rate is unchanged. SPSC-Adaptive avoids needing $\widehat\ell_k$ via
Bernoulli rate $\mu=\Theta((K/T)^{1/3})$; the rate matches the oracle
provided $\mu$ is tuned to this scaling (Thm.~spsc\_adaptive). A
\emph{universal constant} $\mu\in(0,1)$ would produce probe cost
$\Theta(\mu T)$, dominating $\Theta(K^{1/3}T^{2/3})$ by $T^{1/3}$;
the scaling $\mu\asymp(K/T)^{1/3}$ is therefore necessary, not cosmetic.
\end{remark}
